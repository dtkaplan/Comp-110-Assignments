GlyphReady <- BabyNames %>%
  spread(key = sex, value = count) %>% 
  mutate(total = ifelse(is.na(F),0,F) + ifelse(is.na(M),0,M) ) %>% 
  group_by(year)  %>%
  mutate(TheRank = row_number(desc(total)) ) %>%
  mutate(intop100 = (TheRank < 101)) %>% 
  arrange(TheRank) %>% 
  summarise(frac_in_top100 = sum(intop100)/n()) %>% 
  arrange(year)
