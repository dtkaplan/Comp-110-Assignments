count_table <-
  BabyNames %>%
  group_by(year, name) %>%
  summarise(total_count= sum(count)) %>%
  mutate(frac_in_top_100 = rank(desc(total_count)))
count_table$frac_in_top_100 <- ifelse(count_table$frac_in_top_100 <= 100,"TRUE","FALSE")
count_table <-
  count_table %>%
  group_by(year, frac_in_top_100) %>%
  summarise(total= sum(total_count))
GlyphReady <-
  count_table %>%
  spread(frac_in_top_100, total) %>%
  mutate(frac_in_top_100 = `TRUE` / (`TRUE` + `FALSE`))
ggplot(GlyphReady, aes(x=year, y=frac_in_top_100)) + geom_line()
