Translation <- read.csv("http://tiny.cc/dcf/NPT4-names.csv")
SelectData <-
ScorecardSmall %>%
filter(CONTROL !=3)%>%
select(CONTROL, ADM_RATE, NPT41_PUB, NPT43_PUB, NPT45_PUB, NPT41_PRIV, NPT43_PRIV, NPT45_PRIV)

CollegeBoard<-

gather(key=Q135,value=Totalcost, - c(ADM_RATE, CONTROL))%>%
